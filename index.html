GraphMotion is an npm
package that animates GitHub contribution graphs beautifully   
A playful, interactive twist on GitHub's classic heatmap.
Trigger unique animations by clicking letters.
 
"use client"

    import { ContributionCalendar } from 'animated-contribution-heatmap';

    export default function Home() {
    return (
        <div className="h-screen bg-black gap-1 flex flex-col px-10 py-40">
        <h1 className="text-4xl font-bold text-white mb-4">
            Animated contribution Heatmap
        </h1>
        <p className="text-xl text-gray-300 mb-2">
        A beautiful, interactive contribution heatmap calendar with 8 animation patterns
        </p>
        <p className="text-xl text-gray-300 mb-8">
            Click on the letters in "Activity" to trigger different animations!
        </p>
        <ContributionCalendar
            githubToken={process.env.NEXT_PUBLIC_GITHUB_TOKEN}
            username={process.env.NEXT_PUBLIC_GITHUB_USERNAME}
            theme="dark"
        />
        </div>
    );
    }
